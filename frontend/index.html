Backup/Restore my sessions

Your selection is stored in your browser, this app will not store any data outside your computer. To backup your session selection, save the code below:

To restore your session selection, paste your saved code into the below textbox and click "Restore":

  1. Login to the official portal
  2. Open the browser console (F12) and go to "Console" tab
  3. Run the below code to extract your schedule: 
    
var apiProfileToken = null;
var authToken = null;
var widgetId = null;

fetch("https://registration.awsevents.com/flow/loadPage?pageUri=myagenda&workflowApiToken=awsevents.reinvent24.myagenda")
.then(response => response.json())
.then(data => {
    apiProfileToken = data.data.widgetConf.apiProfileToken;
    widgetId = data.data.widgetConf.widgetToken;

    authToken = document.cookie
        .split("; ")
        .find((row) => row.startsWith("rfjwt="))
        ?.split("=")[1];

    fetch("https://catalog.awsevents.com/api/myData", {
        method: "POST",
        credentials: "include",
        headers: {
            "rfapiprofileid": apiProfileToken,
            "rfauthtoken": authToken,
            "rfwidgetid": widgetId
        }
    }).then(response => response.json())
    .then(data => {
        var sessionIds = data.mySchedule.map((session) => session.sessionID);

        var backupCodeObj = {};
        for (let sessionId of sessionIds) {
            backupCodeObj[sessionId] = {"color":"primary"}
        }

        console.log(JSON.stringify(backupCodeObj));
    });
});
  4. Copy the output from the console and paste it into the textbox above
  5. Click "Restore"
